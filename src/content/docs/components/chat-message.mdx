## Usage

```tsx
import { ChatMessage } from "@/components/ui/chat-message"
```

```tsx
<ChatMessage
    key={message.id}
    id={message.id}
    type="outgoing"
    variant="bubble"
>
    <ChatMessageAvatar imageSrc="/avatar-2.png" />
    <ChatMessageContent content={message.content} />
</ChatMessage>
<ChatMessage
    key={message.id}
    id={message.id}
    type="incoming"
    variant="bubble"
>
    <ChatMessageAvatar imageSrc="/avatar-2.png" />
    <ChatMessageContent content={message.content} />
</ChatMessage>
```
