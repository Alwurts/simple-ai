## Usage

```tsx
import { VisualizeTextNode } from "@/components/react-flow/visualize-text-node"
```

The Visualize Text Node component is a React Flow node that allows you to display and edit text content. It includes:

- Editable text area with auto-resizing
- Customizable node styling
- Built-in handle connections
- Support for React Flow's node and edge interactions

```tsx
import { useNodesState } from "reactflow";
import { VisualizeTextNode } from "@/components/react-flow/visualize-text-node";

const initialNodes = [
  {
    id: '1',
    type: 'visualizeText',
    position: { x: 0, y: 0 },
    data: { text: 'Initial text content' },
  },
];

function MyFlow() {
  const [nodes, setNodes] = useNodesState(initialNodes);
  
  return (
    <ReactFlow
      nodes={nodes}
      onNodesChange={setNodes}
      nodeTypes={{ visualizeText: VisualizeTextNode }}
    />
  );
}
```
