## Usage

```tsx
import { ResizableNode } from "@/components/ui/resizable-node"
```

The Resizable Node component is a wrapper around the shadcn/ui Resizable component that allows users to create resizable nodes in React Flow. It includes:

- Smooth resizing handles on all sides
- Minimum and maximum size constraints
- Customizable content area
- Built-in support for React Flow node types

```tsx
import { useState } from "react";
import { ResizableNode } from "@/components/ui/resizable-node";

function MyComponent() {
    const [size, setSize] = useState({ width: 200, height: 150 });
    return (
        <ResizableNode
            size={size}
            onSizeChange={setSize}
            minWidth={100}
            minHeight={100}
            maxWidth={500}
            maxHeight={400}
        >
            <div className="p-4">
                <h3>Resizable Content</h3>
                <p>This area can be resized</p>
            </div>
        </ResizableNode>
    );
}
```
